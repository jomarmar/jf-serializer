62  	private Object getValue(JsonObject obj) throws Exception {
63  		int type = obj.getAsJsonPrimitive("type").getAsInt();
64  		
65  		switch(type) {
66  			case JFDataTypes.TYPE_INTEGER:
67  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
68  				return Integer.parseInt(obj.getAsJsonPrimitive("value").getAsString());
69  				
70  			case JFDataTypes.TYPE_FLOAT:
71  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
72  				return Float.parseFloat(obj.getAsJsonPrimitive("value").getAsString());
73  				
74  			case JFDataTypes.TYPE_LONG:
75  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
76  				return Long.parseLong(obj.getAsJsonPrimitive("value").getAsString());
77  				
78  			case JFDataTypes.TYPE_DOUBLE:
79  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
80  				return Double.parseDouble(obj.getAsJsonPrimitive("value").getAsString());
81  
82  			case JFDataTypes.TYPE_BOOLEAN:
83  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
84  				return Boolean.parseBoolean(obj.getAsJsonPrimitive("value").getAsString());
85  				
86  			case JFDataTypes.TYPE_CHARACTER:
87  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
88  				return new Character(obj.getAsJsonPrimitive("value").getAsString().charAt(0));
89  				
90  			case JFDataTypes.TYPE_BYTE:
91  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
92  				return Byte.parseByte(obj.getAsJsonPrimitive("value").getAsString());
93  				
94  			case JFDataTypes.TYPE_BASE64:
95  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
96  				return Base64.decode(obj.getAsJsonPrimitive("value").getAsString());
97  
98  			case JFDataTypes.TYPE_STRING:
99  				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
100 				return obj.getAsJsonPrimitive("value").getAsString();
101 				
102 			case JFDataTypes.TYPE_DATE:
103 				logger.debug("PRIMITIVE: " + obj.getAsJsonPrimitive("value").getAsString());
104 				SimpleDateFormat format = new SimpleDateFormat(JFDataTypes.DATE_PATTERN);
105 				return format.parse(obj.getAsJsonPrimitive("value").getAsString());
106 				
107 			case JFDataTypes.TYPE_NULL:
108 				logger.debug("PRIMITIVE:  null");
109 				return null;
110 			
111 			case JFDataTypes.TYPE_BOOLEANARRAY:
112 			case JFDataTypes.TYPE_CHARACTERARRAY:
113 			case JFDataTypes.TYPE_INTEGERARRAY:
114 			case JFDataTypes.TYPE_LONGARRAY:
115 			case JFDataTypes.TYPE_FLOATARRAY:
116 			case JFDataTypes.TYPE_DOUBLEARRAY:
117 			case JFDataTypes.TYPE_STRINGARRAY:
118 				logger.debug("ARRAY PRIMITIVE: " + obj.getAsJsonArray("value"));
119 				return getArrayValue(type, obj.getAsJsonArray("value"));
120 				
121 			case JFDataTypes.TYPE_OBJECT:
122 				logger.debug("OBJECT: " + obj.getAsJsonObject("value"));
123 				return getObjectValue(obj.getAsJsonObject("value"));
124 			
125 			case JFDataTypes.TYPE_OBJECTARRAY:
126 				logger.debug("OBJECT ARRAY: " + obj.getAsJsonArray("value"));
127 				return getObjectArrayValue(obj.getAsJsonArray("value"), obj.getAsJsonPrimitive("arrayClass").getAsString());
128 			
129 			case JFDataTypes.TYPE_LIST:
130 			case JFDataTypes.TYPE_VECTOR:
131 				logger.debug("LIST: " + obj.getAsJsonArray("value"));
132 				return getListValue(type, obj.getAsJsonArray("value"));
133 			
134 			case JFDataTypes.TYPE_MAP:
135 			case JFDataTypes.TYPE_TABLE:
136 				logger.debug("MAP: " + obj.getAsJsonArray("value"));
137 				return getMapValue(type, obj.getAsJsonArray("value"));
138 			
139 			default:
140 				return null;

150 		List<Object> list = new ArrayList<Object>();
151 		if(type == JFDataTypes.TYPE_VECTOR) {
152 			list = new Vector<Object>();
153 		}
154 		JFParser parser = new JFParser();
155 		Iterator<JsonElement> iter = arrObj.iterator();
156 		while(iter.hasNext()) {
157 			JsonElement elem = iter.next();
158 			JFParam param = (JFParam) parser.parse(elem.toString());
159 			list.add(param.getValue());
160 			
161 		}
162 		
163 		return list;
164 	}

166 	private Object getMapValue(int type, JsonArray arrObj) {
167 		Map<Object, Object> map = new HashMap<Object, Object>();
168 		if(type == JFDataTypes.TYPE_TABLE) {
169 			map = new Hashtable<Object, Object>();
170 		}
171 		JFParser parser = new JFParser();
172 		Iterator<JsonElement> iter = arrObj.iterator();
173 		while(iter.hasNext()) {
174 			JsonObject elem = iter.next().getAsJsonObject();
175 			JsonObject jkey = elem.getAsJsonObject("key");
176 			JsonObject jval = elem.getAsJsonObject("value");
177 			JFParam paramKey = (JFParam) parser.parse(jkey.toString());
178 			JFParam paramValue = (JFParam) parser.parse(jval.toString());
179 			map.put(paramKey.getValue(), paramValue.getValue());
180 			
181 		}
182 		
183 		return map;
184 	}

188 		int len = arrObj.size();
189 		int i = 0;
190 		Object o = Array.newInstance(Class.forName(arrayType), len);
191 		Iterator<JsonElement> iter = arrObj.iterator();

194 			Object obj = getObjectValue(elem);
195 			Array.set(o, i++, obj);
196 		}

202 	private Object getObjectValue(JsonObject obj) throws Exception {
203 		JFParser parser = new JFParser();
204 		
205 		String cl = obj.getAsJsonPrimitive("clazz").getAsString();
206 		JsonArray array = obj.getAsJsonArray("attr");
207 		List<JFParam> p = new ArrayList<JFParam>();
208 		Iterator<JsonElement> iter = array.iterator();
209 		while(iter.hasNext()) {
210 			JsonElement elem = iter.next();
211 			JFParam param = (JFParam) parser.parse(elem.toString());
212 			p.add(param);
213 		}
214 		return JFSonUtil.getObject(cl, p);
